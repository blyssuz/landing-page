function buildWorkingHoursLine(working_hours) {
    const wh = working_hours || {};
    return DAY_NAMES.map((day, i) => {
        const h = wh[day];
        if (!h || !h.is_open) return `${DAY_LABELS.uz[i]}: yopiq`;
        return `${DAY_LABELS.uz[i]}: ${secsToHHMM(h.start)}-${secsToHHMM(h.end)}`;

}
